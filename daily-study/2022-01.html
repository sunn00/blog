    # 获取数组对象中同一属性的值
    // how do I get [12, 14, 16, 18]
var users = [
  {
    id: 12,
    name: Adam,
  },
  {
    id: 14,
    name: Bob,
  },
  {
    id: 16,
    name: Charlie,
  },
  {
    id: 18,
    name: David,
  },
];

// lodash
_.map(users, "id");

//等同于es5中
users.map((user) => user.id); // [12, 14, 16, 18]
    # 优雅的删除对象中指定属性
    // omit
const object = { a: 1, b: "2", c: 3 };

const result = _.omit(object, ["a", "c"]);
// => { 'b': '2' }

// pick
const object = { a: 1, b: "2", c: 3 };

const result = _.pick(object, ["a", "c"]);
// => { 'a': 1, 'c': 3 }
